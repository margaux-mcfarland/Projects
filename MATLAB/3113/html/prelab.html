% plot temperature at a single point on the rod
%same givens, different fourier term
lam = pi/(2*L);
u = calcTemp(N, bn, lam, x, alpha, t, T0, H);

%varying alphas
alpha = 1:50;
figure(4)
for i = 1:length(alpha)
    %summation
    sum = 0;
    for j = 1:N
        sum = sum + bn*sin(lam.*x)*exp(-lam^2*alpha(i).*t);
    end
    %add steady state and transient state
    u = T0 + H*x + sum;
    plot(t, u);
    hold on;
end
xlabel('time (s)');
ylabel('Temperature (deg C)');
title('Temperature vs time with varying thermal diffusivities');

function u = calcTemp(N, bn, lam, x, alpha, t, T0, H, hold)
% this function calculates the temperature over time of the rod. It also
% plots the temperature over time
%summation
sum = 0;
for i = 1:N
    sum = sum + bn*sin(lam.*x)*exp(-lam^2*alpha.*t);
end

%add steady state and transient state
u = T0 + H*x + sum;

%plot
figure
plot(t, u);
grid on
xlabel("Time (s)");
ylabel("Temperature (deg C)");
str = sprintf("Temperature at a point on the rod vs time (up to %fs)",max(t));
title(str);

end
